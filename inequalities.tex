\section{Walkthrough Problems}

\begin{problem}
If $abcd$ = 1 for $a$, $b$, $c$, $d > 0$, prove that
\[
a^4b + b^4c + c^4d + d^4a \ge a + b + c + d.
\]
\end{problem}

\begin{proof}
    By Hölder, we have:
    \[
    \left( \sum_{\text{cyc}} a^4b \right) \left( \sum_{\text{cyc}} a \right) \left( \sum_{\text{cyc}} c \right) \left( \sum_{\text{cyc}} d \right)
    \ge
    \left( \sum_{\text{cyc}} \sqrt[4]{a^4 abcd} \right)^4
    \]
    \[
    \implies \left( \sum_{\text{cyc}} a^4b \right) (a+b+c+d)^3 \ge (a+b+c+d)^4
    \]
    And we're done.
\end{proof}

\begin{problem}
  [IMO 2001]
  Let $a$, $b$, $c$ be positive reals. Prove that
  \[
  \frac{a}{\sqrt{a^2 + 8bc}} + \frac{b}{\sqrt{b^2 + 8ca}} + \frac{c}{\sqrt{c^2 + 8ab}} \ge 1.
  \]
\end{problem}

\begin{proof}
  Use Hölder to get the following:
  \begin{align*}
    \left( \sum_{\text{cyc}} \frac{a}{\sqrt{a^2 + 8bc}} \right)^2 \left( \sum_{\text{cyc}} a(a^2 + 8bc) \right)
    &\ge
    \left( \sum_{\text{cyc}} \sqrt[3]{\frac{a^2}{a^2 + 8bc} \cdot a(a^2 + 8bc)} \right)^3 \\
    &= \left( \sum_{\text{cyc}} a \right)^3
  \end{align*}
  However we note:
  \[
  \left( \sum_{\text{cyc}} a \right)^3 = (a + b + c)^3 \ge a^3 + b^3 + c^3 + 24abc = \left( \sum_{\text{cyc}} a(a^2 + 8bc) \right)
  \]
  So we're done.
\end{proof}

\begin{problem}
  [IMO Shortlist 2009]
  Let $a$, $b$, $c$ be positive real numbers such that $\frac1a + \frac1b + \frac1c = a + b + c$. Prove that:
  \[
  \frac{1}{(2a+b+c)^2} + \frac{1}{(a+2b+c)^2} + \frac{1}{(aa+b+2c)^2} \le \frac{3}{16}.
  \]
\end{problem}

\begin{proof}
  We begin by rewriting.
  \[
  \sum_{\text{cyc}} \frac{1}{(2a+b+c)^2} \le \frac{3}{16}
  \]
  \[
  \sum_{\text{cyc}} \frac{16}{(2a+b+c)^2} \le 3
  \]
  \[
  \sum_{\text{cyc}} \frac{1}{a(a+b+c)} = 1 \implies \sum_{\text{cyc}} \left( \frac{16}{(2a+b+c)^2} - \frac{3}{a(a+b+c)} \right) \le 0
  \]
  Which means the inequality is homogenized. WLOG let $a + b + c = 3$, then we have
  \[
  \sum_{\text{cyc}} \left( \frac{16}{(3+a)^2} - \frac1a \right) \le 0
  \]
  \[
  \text{Define } f(x) = \frac{16}{(x+3)^2} - \frac1x \text{ then compute } f''(x) = \frac{96}{(x+3)^4} - \frac{2}{x^3}.
  \]
  Note that $f$ is concave on $[0, 3]$. Therefore, by Jensen,
  \[
  \frac{f(a) + f(b) + f(c)}{3} \le 3f\left(\frac{a+b+c}{3}\right) = 3f(1) = 0 \implies \boxed{f(a) + f(b) + f(c) \le 0}.
  \]
\end{proof}

\section{Problems}

\begin{problem}
  [Canadian Olympiad 2002] Let $a$, $b$, $c$ be positive reals. Prove that
  \[
  \frac{a^3}{bc} + \frac{b^3}{ca} + \frac{c^3}{ab} \ge a + b + c.
  \]
\end{problem}

\begin{proof}
  By AM-GM, we have:
  \[
  \frac{a^3}{bc} + b + c \ge 3a
  \]
  and summing cyclically and simplifying gives our result.
\end{proof}
